// ========================================== // PROTEKSI LOGIN - SEDERHANA // ========================================== (function() { console.log('Checking authentication...'); // Cek session di localStorage const sessionData = localStorage.getItem('sb-session'); if (!sessionData) { console.log('No session found, redirecting to login...'); window.location.href = 'login.html'; return; } try { const session = JSON.parse(sessionData); if (!session || !session.user) { console.log('Invalid session, redirecting to login...'); window.location.href = 'login.html'; return; } console.log('User logged in:', session.user.email); // Tampilkan nama user const userName = session.user.user_metadata?.name || session.user.email || 'Mahasiswa'; // Update navbar setelah halaman selesai loading setTimeout(() => { const userBtn = document.querySelector('.fa-user-circle')?.parentElement; if (userBtn) { userBtn.innerHTML = `
${userName.charAt(0).toUpperCase()}
${userName}
`; userBtn.style.cursor = 'pointer'; userBtn.onclick = function() { if (confirm('Apakah Anda yakin ingin keluar?')) { logout(); } }; } // Tambahkan tombol logout const navRight = document.querySelector('.flex.items-center.gap-4'); if (navRight) { const existingLogout = navRight.querySelector('.logout-btn'); if (!existingLogout) { const logoutBtn = document.createElement('button'); logoutBtn.className = 'logout-btn w-10 h-10 rounded-full bg-white/5 hover:bg-red-500/20 transition-all flex items-center justify-center text-white/60 hover:text-red-400 border border-white/5'; logoutBtn.innerHTML = ''; logoutBtn.title = 'Keluar'; logoutBtn.onclick = function() { if (confirm('Apakah Anda yakin ingin keluar?')) { logout(); } }; navRight.appendChild(logoutBtn); } } }, 500); } catch (error) { console.error('Session parse error:', error); window.location.href = 'login.html'; } })(); // Fungsi logout function logout() { localStorage.removeItem('sb-user'); localStorage.removeItem('sb-session'); window.location.href = 'login.html'; }
